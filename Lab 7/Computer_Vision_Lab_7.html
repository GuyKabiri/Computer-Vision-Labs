class FocalLoss(nn.Module):
    def __init__(self, alpha=0.2, gamma=2, logist=False, reduce='mean'):
        super(FocalLoss, self).__init__()
        self.alpha = alpha # the scalar factor between 0 and 1 
        self.gamma = gamma # focusing parameter(always positive) that reduces the relative loss for well-classified examples and puts more focus on hard misclassified examples
        self.cross_entropy_loss = nn.CrossEntropyLoss()
        #self.logist = logist # log probabilities 
        self.reduce = reduce # Specifies the reduction to apply to the output - none/mean/sum. ‘none’: no reduction will be applied, ‘mean’: the sum of the output will be divided by the number of elements in the output, ‘sum’: the output will be summed.

    def forward(self, inputs, targets):

        BCE_loss = self.cross_entropy_loss(inputs, targets)
        pt = torch.exp(-BCE_loss)
        F_loss = self.alpha * (1-pt)**self.gamma * BCE_loss

        if self.reduce:
            return torch.mean(F_loss)
        else:
            return F_loss

class AverageMeter(object):
    """Computes and stores the average and current value"""
    def __init__(self):
        self.reset()

    def reset(self):
        self.val = 0
        self.avg = 0
        self.sum = 0
        self.count = 0

    def update(self, val, n=1):
        self.val = val
        self.sum += val * n
        self.count += n
        self.avg = self.sum / self.count

class MNISTDataset(Dataset):
    def __init__(self, data, labels, transforms=None):
        super(MNISTDataset, self).__init__()

        if len(data) != len(labels):
            raise ValueError("The size of `data` and `labels` should be equals")

        self.x = data
        self.y = labels
        self.transforms = transforms

    def __getitem__(self, idx):
        img = self.x[idx]
        lbl = self.y[idx]
        
        if self.transforms is not None:
            img = self.transforms(image=img)['image']
        
        img = img[np.newaxis, :, :]
        return img, lbl

    def __len__(self):
        return len(self.x)

# LeNet skeleton!
class LeNet(nn.Module):
    def __init__(self):
        super(LeNet, self).__init__()
        self.conv1 = nn.Conv2d(1, 6, 5)
        self.pool1 = nn.MaxPool2d(2)
        self.conv2 = nn.Conv2d(6, 16, 5)
        self.pool2 = nn.MaxPool2d(2)
        self.fc1 = nn.Linear(256, 120)
        self.fc2 = nn.Linear(120, 84)
        self.fc3 = nn.Linear(84, 10)
        

    def forward(self, x):
        x = F.relu(self.conv1(x))
        x = self.pool1(x)
        x = F.relu(self.conv2(x))
        x = self.pool2(x)
        x = x.view(x.shape[0], -1)
        x = F.relu(self.fc1(x))
        x = F.relu(self.fc2(x))
        x = self.fc3(x)
        return x

Train Function¶

Single train over the whole data (epoch)

In [ ]:
'''
    dataloader  -   a torch dataloader object
    model       -   a torch nn.Module object
    criterion   -   loss function
    score       -   score function
    optimizer   -   a torch optimizer object
'''
def train(dataloader, model, criterion, score, optimizer):
    accuracies = AverageMeter()
    losses = AverageMeter()
    model.train()

    for x_batch, y_batch in dataloader:
        optimizer.zero_grad()       #   reset grads

        output = model(x_batch.float())     #   evaluate outputs

        curr_loss = criterion(output, y_batch.long())       #   calculate loss

        curr_loss.backward()                                #   backpropthe loss
        losses.update(curr_loss.item(), x_batch.size(0))    #   save the current loss
        optimizer.step()                                    

        curr_accuracy = score(output.argmax(dim=-1), y_batch)     # dim=-1   ->   -1 is the last dimentions
        accuracies.update(curr_accuracy, x_batch.size(0))   #   save current accuracy

    return losses.avg, accuracies.avg

'''
    dataloader  -   a torch dataloader object
    model       -   a torch nn.Module object
    criterion   -   loss function
    score       -   score function
'''
def validation(dataloader, model, criterion, score):
    accuracies = AverageMeter()
    losses = AverageMeter()
    model.eval()
    
    for x_batch, y_batch in dataloader:

        output = model(x_batch.float())     #   evaluate outputs

        curr_loss = criterion(output, y_batch.long())       #   calculate loss
        losses.update(curr_loss.item(), x_batch.size(0))    #   save the current loss

        curr_accuracy = score(output.argmax(dim=-1), y_batch)     # dim=-1   ->   -1 is the last dimentions
        accuracies.update(curr_accuracy, x_batch.size(0))

    return losses.avg, accuracies.avg

Full Training Function¶

Train the model over the whole training dataset for a selected number of epochs.
Run validation at the end of each training epoch, and saving the loss and accuracy scores of the training and validation.

In [ ]:
'''
    model               -   a torch nn.Module object
    train_dataloader    -   training dataloader
    valid_dataloader    -   validation dataloader
    criterion           -   loss function
    optimizer           -   optimizer object for training
    score               -   score function
    epochs              -   number of epochs
    print_freq          -   frequency of printing
'''
def full_training(model, train_dataloader, valid_dataloader, criterion, optimizer, score, epochs, print_freq=1):
    train_loss_arr = []
    train_acc_arr = []

    min_valid_loss = np.inf     

    val_loss_arr = []
    val_score_arr = []

    model_best_params = None

    scores_losses_df = pd.DataFrame(columns=['epoch', 'split', 'loss', 'score'])

    for e in range(epochs):
        train_loss, train_score = train(train_dataloader, model, criterion, score, optimizer)   #   single train over the whole data
        train_loss_arr.append(train_loss)
        train_acc_arr.append(train_score)

        valid_loss, valid_score = validation(valid_dataloader, model, criterion, score)
        val_loss_arr.append(valid_loss)
        val_score_arr.append(valid_score)

        scores_losses_df.loc[len(scores_losses_df)] = [e, 'train', train_loss, train_score]     #   save results of training
        scores_losses_df.loc[len(scores_losses_df)] = [e, 'valid', valid_loss, valid_score]     #   save results of validation


        #   if validation loss is better than before -> save model state dictionary
        if min_valid_loss > valid_loss:
            min_valid_loss = valid_loss
            # torch.save(model.state_dict(), 'saved_model.pth')
            model_best_params = model.state_dict()

        if print_freq is not None and e % print_freq == 0:
            print('Epoch: {:4d}, '
                    'Train_acc: {:.5f}, Train_loss: {:.5f}, '
                    'Valid_acc: {:.5f}, Valid_loss: {:.5f}'
                    .format(e,
                            train_score,
                            train_loss,
                            valid_score,
                            valid_loss))
            
    return scores_losses_df, model_best_params

class Mnist100Data:
    def __init__(self, data):
        self.images = []
        self.labels = []

        for (img, lbl) in data:
            self.images.append(img)
            self.labels.append(lbl)
        
        self.images = np.array(self.images)
        self.labels = np.array(self.labels)


    def load_data(self, train_size=.9, shuffle=True):

        if train_size > 1 or train_size < 0:
            raise ValueError("Split sizes can not be greater than 1 or less than 0")

        test_size = 1 - train_size
        x_train, x_test, y_train, y_test = train_test_split(self.images, self.labels, test_size=test_size, random_state=SEED)

        return (x_train, y_train), (x_test, y_test)

class FocalLoss2(nn.modules.loss._WeightedLoss):
    def __init__(self, alpha=None, gamma=2, reduction='mean'):
        super(FocalLoss2, self).__init__(weight=alpha,reduction=reduction)
        self.gamma = gamma
        self.alpha = alpha

    def forward(self, input, target):

        ce_loss = F.cross_entropy(input, target, reduction=self.reduction, weight=self.alpha) 
        pt = torch.exp(-ce_loss)
        focal_loss = ((1 - pt) ** self.gamma * ce_loss).mean()
        return focal_loss

def calc_layer_shape(input_dim, filter_shape, stride=1):
    return int((input_dim - filter_shape) / stride) + 1

class LeNet3D(nn.Module):
    '''
        `input_shape`   -   should as follow: [in_channel, in_height, in_width]
        `output_dim`    -   is the number of prediction in the output layer
    '''
    def __init__(self, input_shape, output_dim):
        super(LeNet3D, self).__init__()
        c, h, w = input_shape

        self.conv1 = nn.Conv2d(c, 6, 5)
        c, h, w = 6, calc_layer_shape(h, 5), calc_layer_shape(w, 5)

        self.pool1 = nn.MaxPool2d(2)
        c, h, w = 6, calc_layer_shape(h, 2, 2), calc_layer_shape(w, 2, 2)

        self.conv2 = nn.Conv2d(6, 16, 5)
        c, h, w = 16, calc_layer_shape(h, 5), calc_layer_shape(w, 5)

        self.pool2 = nn.MaxPool2d(2)
        c, h, w = 16, calc_layer_shape(h, 2, 2), calc_layer_shape(w, 2, 2)

        flatten_shape = c * h * w

        self.fc1 = nn.Linear(flatten_shape, 120)
        self.fc2 = nn.Linear(120, 84)
        self.fc3 = nn.Linear(84, output_dim)
        

    def forward(self, x):
        x = F.relu(self.conv1(x))
        x = self.pool1(x)
        x = F.relu(self.conv2(x))
        x = self.pool2(x)
        x = x.view(x.shape[0], -1)
        x = F.relu(self.fc1(x))
        x = F.relu(self.fc2(x))
        x = self.fc3(x)
        return x
